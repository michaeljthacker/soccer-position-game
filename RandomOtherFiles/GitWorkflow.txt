# Stage all changes
git add .

# Commit the changes with a message
git commit -m "Your commit message"

# Push the changes to the remote branch
git push origin branchname

# Create a Pull Request on GitHub
# - Go to your repository on GitHub
# - Navigate to the "Pull requests" tab
# - Click "New pull request"
# - Select the branch you pushed to
# - Create the pull request
# - Merge the pull request once it has been reviewed
# - Delete the branch on GitHub after merging the pull request.

# Fetch the latest changes from the remote repository
git fetch origin

# Switch to the main branch (or the branch you merged into)
git checkout main

# Pull the latest changes from the remote main branch
git pull origin main

# Delete the local branch that you merged and deleted on GitHub
git branch -d branchname

# Prune remote-tracking branches that no longer exist on the remote
git fetch --prune
